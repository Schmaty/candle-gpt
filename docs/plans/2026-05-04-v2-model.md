# Candle-GPT v2 — Model Architecture (Plan 3 of 5)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Define `CandleGPTv2`, a ~10.9M-parameter causal transformer that takes a (batch, seq_len, 41) float32 feature tensor and outputs a (batch, seq_len, 256) logit distribution over discretized next-bar log-return bins. Also define `ReturnTokenizerV2` for quantile-based bin fitting, encode, decode, and persistence. **No training in this plan** — only architecture, forward pass, and tests.

**Architecture:** Continuous-input GPT-style transformer. Input features are linearly projected from 41 → d_model=384. A learned positional embedding is added. Six causal transformer blocks (6-head attention, 1536-dim FFN, pre-LN). A final LayerNorm + linear head maps to N_BINS=256 logits. The tokenizer uses quantile binning — it is fitted externally (Plan 4) and saved to disk; the model itself is token-agnostic.

**Tech Stack:** Python 3.11+, torch, numpy, pytest. No new pip deps needed.

**Parameter budget:**
- Input projection: 41×384 = 15,744
- Positional embedding: 512×384 = 196,608
- 6 transformer blocks × ~1,772,160 = 10,632,960
- Output LN + head: 768 + 384×256 = 99,072
- **Total: ≈10,944,384 (~10.9M)**

```python
"""Model hyperparameter configuration for CandleGPTv2."""
from __future__ import annotations
from dataclasses import dataclass, field, asdict
from typing import Any


@dataclass
class ModelConfig:
    n_features: int = 41          # feature vector width (from v2.features.constants)
    d_model: int = 384            # transformer hidden dimension
    n_heads: int = 6              # attention heads (head_dim = d_model / n_heads = 64)
    n_layers: int = 6             # transformer blocks
    ffn_mult: int = 4             # FFN hidden = d_model * ffn_mult
    block_size: int = 512         # maximum sequence length
    n_bins: int = 256             # return vocabulary size (softmax classes)
    dropout: float = 0.1          # applied in attention + FFN

    def __post_init__(self) -> None:
        if self.d_model % self.n_heads != 0:
            raise ValueError(
                f"head_dim = d_model/n_heads = {self.d_model}/{self.n_heads} "
                f"is not an integer"
            )

    @property
    def ffn_dim(self) -> int:
        return self.d_model * self.ffn_mult

    @property
    def head_dim(self) -> int:
        return self.d_model // self.n_heads

    def to_dict(self) -> dict[str, Any]:
        return asdict(self)

    @classmethod
    def from_dict(cls, d: dict[str, Any]) -> "ModelConfig":
        return cls(**{k: v for k, v in d.items() if k in cls.__dataclass_fields__})
```

```python
"""Causal multi-head self-attention with rotary-free absolute position bias.

Uses a standard additive causal mask (upper-triangle = -inf). Compatible with
torch.compile and MPS (no flash-attn dependency).
"""
from __future__ import annotations
import math

import torch
import torch.nn as nn
import torch.nn.functional as F

from v2.model.config import ModelConfig


class CausalSelfAttention(nn.Module):
    def __init__(self, cfg: ModelConfig) -> None:
        super().__init__()
        self.cfg = cfg
        self.qkv = nn.Linear(cfg.d_model, 3 * cfg.d_model, bias=False)
        self.proj = nn.Linear(cfg.d_model, cfg.d_model, bias=False)
        self.attn_drop = nn.Dropout(cfg.dropout)
        self.resid_drop = nn.Dropout(cfg.dropout)
        # Causal mask — upper triangle filled with -inf, lower (incl diag) = 0.
        self.register_buffer(
            "mask",
            torch.triu(
                torch.full((cfg.block_size, cfg.block_size), float("-inf")),
                diagonal=1,
            ),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        B, T, C = x.shape
        assert C == self.cfg.d_model
        q, k, v = self.qkv(x).split(C, dim=2)
        h = self.cfg.n_heads
        q = q.view(B, T, h, C // h).transpose(1, 2)
        k = k.view(B, T, h, C // h).transpose(1, 2)
        v = v.view(B, T, h, C // h).transpose(1, 2)

        scale = 1.0 / math.sqrt(k.size(-1))
        att = (q @ k.transpose(-2, -1)) * scale
        att = att + self.mask[:T, :T]
        att = F.softmax(att, dim=-1)
        att = self.attn_drop(att)
        y = att @ v
        y = y.transpose(1, 2).contiguous().view(B, T, C)
        return self.resid_drop(self.proj(y))
```

```python
"""Pre-LN transformer block: LayerNorm → Attention → residual, LayerNorm → FFN → residual."""
from __future__ import annotations

import torch
import torch.nn as nn
import torch.nn.functional as F

from v2.model.attention import CausalSelfAttention
from v2.model.config import ModelConfig


class TransformerBlock(nn.Module):
    def __init__(self, cfg: ModelConfig) -> None:
        super().__init__()
        self.ln1 = nn.LayerNorm(cfg.d_model)
        self.attn = CausalSelfAttention(cfg)
        self.ln2 = nn.LayerNorm(cfg.d_model)
        self.ffn = nn.Sequential(
            nn.Linear(cfg.d_model, cfg.ffn_dim, bias=False),
            nn.GELU(),
            nn.Linear(cfg.ffn_dim, cfg.d_model, bias=False),
            nn.Dropout(cfg.dropout),
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = x + self.attn(self.ln1(x))
        x = x + self.ffn(self.ln2(x))
        return x
```

```python
"""CandleGPTv2: continuous-input causal transformer for next-bar return prediction.

Input:  (batch, seq_len, n_features=41) float32
Output: (batch, seq_len, n_bins=256) float32 logits

Forward pass:
  1. Linear projection: n_features → d_model
  2. Add learned positional embedding (shape: block_size × d_model)
  3. N transformer blocks (pre-LN, causal attention)
  4. LayerNorm
  5. Linear head: d_model → n_bins

At position t, logits[t] predicts the bin ID of log(close[t+1] / close[t]).
"""
from __future__ import annotations

import torch
import torch.nn as nn
import torch.nn.functional as F

from v2.model.config import ModelConfig
from v2.model.transformer import TransformerBlock


class CandleGPTv2(nn.Module):
    def __init__(self, cfg: ModelConfig) -> None:
        super().__init__()
        self.cfg = cfg
        self.input_proj = nn.Linear(cfg.n_features, cfg.d_model, bias=False)
        self.pos_embed = nn.Embedding(cfg.block_size, cfg.d_model)
        self.drop = nn.Dropout(cfg.dropout)
        self.blocks = nn.ModuleList([TransformerBlock(cfg) for _ in range(cfg.n_layers)])
        self.ln_f = nn.LayerNorm(cfg.d_model)
        self.head = nn.Linear(cfg.d_model, cfg.n_bins, bias=False)
        self._init_weights()

    def _init_weights(self) -> None:
        for module in self.modules():
            if isinstance(module, nn.Linear):
                nn.init.normal_(module.weight, mean=0.0, std=0.02)
            elif isinstance(module, nn.Embedding):
                nn.init.normal_(module.weight, mean=0.0, std=0.02)
            elif isinstance(module, nn.LayerNorm):
                nn.init.ones_(module.weight)
                nn.init.zeros_(module.bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """x: (B, T, n_features) → logits: (B, T, n_bins)"""
        B, T, _ = x.shape
        assert T <= self.cfg.block_size, \
            f"Sequence length {T} exceeds block_size {self.cfg.block_size}"
        pos = torch.arange(T, device=x.device)
        h = self.drop(self.input_proj(x) + self.pos_embed(pos))
        for block in self.blocks:
            h = block(h)
        h = self.ln_f(h)
        return self.head(h)

    def num_params(self, exclude_pos_embed: bool = False) -> int:
        total = sum(p.numel() for p in self.parameters())
        if exclude_pos_embed:
            total -= self.pos_embed.weight.numel()
        return total

    @torch.no_grad()
    def generate_ids(
        self,
        x: torch.Tensor,
        n_steps: int,
        temperature: float = 1.0,
        top_k: int | None = None,
    ) -> torch.Tensor:
        """Autoregressively sample n_steps bin IDs given a context tensor x.

        x: (B, T, n_features) — the conditioning context.
        Returns: (B, n_steps) int64 bin IDs.

        Note: This is a stub for autoregressive generation used in inference and
        evaluation. During training, the model is called once on the full sequence
        and the loss is computed at every position simultaneously (teacher forcing).
        For single-step prediction (the dashboard's live view), call forward() on
        the context and argmax/sample from the last position's logits directly.
        """
        self.eval()
        B = x.size(0)
        out_ids = []
        ctx = x  # (B, T, n_features)
        for _ in range(n_steps):
            T = ctx.size(1)
            ctx_crop = ctx[:, -self.cfg.block_size:, :]
            logits = self.forward(ctx_crop)  # (B, T, n_bins)
            logits_last = logits[:, -1, :] / max(temperature, 1e-8)  # (B, n_bins)
            if top_k is not None:
                v, _ = torch.topk(logits_last, min(top_k, logits_last.size(-1)))
                logits_last[logits_last < v[:, [-1]]] = float("-inf")
            probs = F.softmax(logits_last, dim=-1)
            ids = torch.multinomial(probs, num_samples=1)  # (B, 1)
            out_ids.append(ids)
            # For the next step we need a feature vector for the sampled bar.
            # Since we don't have actual future OHLCV data, we reuse the last
            # context bar as a placeholder. In real inference, the caller
            # reconstructs the bar from the sampled return and calls forward()
            # with the updated feature vector.
            ctx = torch.cat([ctx, ctx[:, -1:, :]], dim=1)
        return torch.cat(out_ids, dim=1)
```

```python
    # Within 3× the typical bin width at that quantile
```

```python
"""ReturnTokenizerV2: quantile-based discretization of log-return values.

Usage:
    tok = ReturnTokenizerV2(n_bins=256)
    tok.fit(training_log_returns)      # numpy float64 array
    bin_ids = tok.encode(log_returns)  # int64 numpy array, shape (N,)
    log_rets = tok.decode(bin_ids)     # float64 numpy array (bin midpoints)
    tok.save(path)
    tok2 = ReturnTokenizerV2.load(path)

Fitting is done at the start of Plan 4 on the training split only. The fitted
tokenizer is saved alongside training checkpoints so inference and eval use the
same bin boundaries.
"""
from __future__ import annotations
import pickle
from pathlib import Path

import numpy as np


class ReturnTokenizerV2:
    """Quantile-based tokenizer for log-return values.

    Bins are determined by equal-frequency quantile cuts over the training data,
    so each bin covers approximately the same fraction of the probability mass.
    This avoids the extreme-return bins being arbitrarily wide.
    """

    def __init__(self, n_bins: int = 256) -> None:
        self._n_bins = n_bins
        self.breakpoints: np.ndarray | None = None

    @property
    def n_bins(self) -> int:
        return self._n_bins

    def fit(self, log_returns: np.ndarray) -> None:
        """Compute n_bins+1 quantile breakpoints from training log returns.

        Infinite and NaN values are removed before fitting.
        If quantile collapse reduces the effective bin count (duplicate quantile
        values), n_bins is reduced to the actual number of unique breakpoints - 1.
        """
        data = np.asarray(log_returns, dtype=np.float64)
        data = data[np.isfinite(data)]
        q = np.linspace(0.0, 1.0, self._n_bins + 1)
        breaks = np.quantile(data, q)
        breaks = np.unique(breaks)  # remove duplicates from flat regions
        self._n_bins = len(breaks) - 1
        self.breakpoints = breaks

    def encode(self, log_returns: np.ndarray) -> np.ndarray:
        """Map log returns to bin IDs in [0, n_bins-1].

        Values below breakpoints[0] → bin 0.
        Values above breakpoints[-1] → bin n_bins-1.
        """
        if self.breakpoints is None:
            raise RuntimeError("Tokenizer is not fitted; call fit() first.")
        data = np.asarray(log_returns, dtype=np.float64)
        # np.digitize: returns 0 for data < breaks[0], len(breaks) for data > breaks[-1]
        # We use breaks[1:-1] as interior edges so all data maps to [0, n_bins-1].
        ids = np.digitize(data, self.breakpoints[1:-1])
        return np.clip(ids, 0, self._n_bins - 1).astype(np.int64)

    def decode(self, bin_ids: np.ndarray) -> np.ndarray:
        """Map bin IDs back to the midpoint of each bin in log-return space."""
        if self.breakpoints is None:
            raise RuntimeError("Tokenizer is not fitted; call fit() first.")
        centers = (self.breakpoints[:-1] + self.breakpoints[1:]) / 2.0
        ids = np.clip(np.asarray(bin_ids, dtype=np.int64), 0, len(centers) - 1)
        return centers[ids]

    def save(self, path: Path) -> None:
        path = Path(path)
        path.parent.mkdir(parents=True, exist_ok=True)
        with open(path, "wb") as f:
            pickle.dump(self, f, protocol=pickle.HIGHEST_PROTOCOL)

    @classmethod
    def load(cls, path: Path) -> "ReturnTokenizerV2":
        with open(Path(path), "rb") as f:
            obj = pickle.load(f)
        if not isinstance(obj, cls):
            raise TypeError(f"Expected ReturnTokenizerV2, got {type(obj)}")
        return obj
```

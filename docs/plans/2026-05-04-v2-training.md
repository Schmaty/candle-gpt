# Candle-GPT v2 — Training Run (Plan 4 of 5)

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Train `CandleGPTv2` on BTCUSDT 1m data for up to 6 hours wall-clock, checkpoint every 30 minutes, evaluate the best validation checkpoint on held-out 2025-Q4 + 2026 data, and write a `REPORT.md` with calibration + per-regime accuracy + sample predictions. Training must survive this Claude session exiting (launched via `nohup`). Total wall-clock cap is hard-coded: the training loop checks elapsed time and exits gracefully.

**Architecture:** Three Python modules under `v2/train/`:
- `config.py` — `TrainConfig` dataclass (all hyperparameters, paths, split boundaries)
- `loop.py` — `train(cfg)` function: dataset splits, tokenizer fit, AdamW + cosine LR, training loop with wall-clock guard, periodic checkpoint saves
- `eval.py` — `evaluate(cfg, ckpt_path)` → metrics dict + `write_report(metrics, run_dir)`
- `run.py` — CLI entry point (`python -m v2.train.run`) that calls `train()` then `evaluate()`; designed to run under `nohup`

**Data split (pre-computed, stable):**
- Training: first 72% of bars (≈ 2022-01 to mid-2024)
- Validation: next 12% (≈ mid-2024 to early 2025)
- Test / held-out: last 16% (≈ 2025-Q4 + 2026)

The split is done by bar-index count (not by calendar date) to match the 72/12/16 regime column distribution already on disk.

**Tech Stack:** Python 3.11+, torch (MPS preferred on Apple Silicon, CPU fallback), AdamW, CosineAnnealingLR, uv for env, nohup for process persistence.

```python
"""Training hyperparameters and path configuration for CandleGPTv2."""
from __future__ import annotations
import time
from dataclasses import dataclass, field
from pathlib import Path

from v2.model.config import ModelConfig


@dataclass
class TrainConfig:
    # --- Paths ---
    raw_dir: Path = field(default_factory=lambda: Path("v2/data/raw"))
    runs_dir: Path = field(default_factory=lambda: Path("v2/runs"))
    run_id: str = field(default_factory=lambda: time.strftime("%Y%m%d_%H%M%S"))

    # --- Model ---
    model: ModelConfig = field(default_factory=ModelConfig)

    # --- Data ---
    kline_file: str = "btcusdt_1m.parquet"
    funding_file: str = "funding_btcusdt.parquet"
    liq_file: str = "liq_btcusdt_per_minute.parquet"
    window: int = 512          # sequence length fed to the model
    stride_train: int = 64     # sliding window stride for training dataset
    stride_val: int = 512      # non-overlapping windows for val/test

    # --- Split (by bar index) ---
    train_frac: float = 0.72
    val_frac: float = 0.12
    # test_frac = 1 - train_frac - val_frac = 0.16 (implicit)

    # --- Optimiser ---
    lr_max: float = 3e-4
    lr_min: float = 3e-5       # cosine LR decays to this floor
    weight_decay: float = 0.1
    beta1: float = 0.9
    beta2: float = 0.95
    grad_clip: float = 1.0
    batch_size: int = 32

    # --- Schedule ---
    warmup_steps: int = 500
    max_steps: int = 200_000   # upper bound; wall-clock cap typically fires first

    # --- Wall-clock cap ---
    max_wall_clock_s: float = 6 * 3600.0   # 6 hours, hard stop
    checkpoint_interval_s: float = 30 * 60.0  # checkpoint every 30 min

    # --- Eval ---
    val_interval_steps: int = 500    # evaluate on val set every N steps
    val_batches: int = 100           # how many val batches per evaluation pass
    log_interval_steps: int = 50     # print training loss every N steps

    # --- Tokenizer ---
    n_bins: int = 256  # must match model.n_bins

    @property
    def run_dir(self) -> Path:
        return self.runs_dir / self.run_id

    @property
    def ckpt_dir(self) -> Path:
        return self.run_dir / "checkpoints"

    @property
    def kline_path(self) -> Path:
        return self.raw_dir / self.kline_file

    @property
    def funding_path(self) -> Path:
        return self.raw_dir / self.funding_file

    @property
    def liq_path(self) -> Path:
        return self.raw_dir / self.liq_file

    @property
    def tokenizer_path(self) -> Path:
        return self.run_dir / "tokenizer.pkl"

    @property
    def best_ckpt_path(self) -> Path:
        return self.ckpt_dir / "best_val.pt"

    @property
    def report_path(self) -> Path:
        return self.run_dir / "REPORT.md"
```

```python
"""Training loop for CandleGPTv2.

train(cfg) performs:
1. Fit ReturnTokenizerV2 on training split log returns.
2. Build train/val/test KlineWindowDataset slices (by bar-index).
3. Run AdamW + cosine LR loop with wall-clock cap.
4. Checkpoint every cfg.checkpoint_interval_s seconds + on best val loss.
5. Return the run_id string.
"""
from __future__ import annotations
import json
import logging
import math
import time
from pathlib import Path
from typing import Iterator

import numpy as np
import torch
import torch.nn as nn
from torch.utils.data import DataLoader, Subset

from v2.data.dataset import KlineWindowDataset
from v2.data.store import read_klines
from v2.features.engineer import compute_features
from v2.model.config import ModelConfig
from v2.model.model import CandleGPTv2
from v2.model.tokenizer import ReturnTokenizerV2
from v2.train.config import TrainConfig

log = logging.getLogger(__name__)


def _select_device() -> torch.device:
    if torch.backends.mps.is_available():
        return torch.device("mps")
    if torch.cuda.is_available():
        return torch.device("cuda")
    return torch.device("cpu")


def _get_lr(step: int, cfg: TrainConfig) -> float:
    """Linear warmup then cosine decay from lr_max to lr_min."""
    if step < cfg.warmup_steps:
        return cfg.lr_max * step / max(cfg.warmup_steps, 1)
    progress = (step - cfg.warmup_steps) / max(cfg.max_steps - cfg.warmup_steps, 1)
    cosine_decay = 0.5 * (1.0 + math.cos(math.pi * min(progress, 1.0)))
    return cfg.lr_min + cosine_decay * (cfg.lr_max - cfg.lr_min)


def _save_checkpoint(
    model: CandleGPTv2,
    optimizer: torch.optim.Optimizer,
    step: int,
    val_loss: float,
    cfg: TrainConfig,
    tag: str,
) -> Path:
    cfg.ckpt_dir.mkdir(parents=True, exist_ok=True)
    path = cfg.ckpt_dir / f"{tag}.pt"
    torch.save(
        {
            "step": step,
            "val_loss": val_loss,
            "model_state": model.state_dict(),
            "optimizer_state": optimizer.state_dict(),
            "model_config": cfg.model.to_dict(),
            "run_id": cfg.run_id,
            "tokenizer_path": str(cfg.tokenizer_path),
        },
        path,
    )
    log.info(f"Checkpoint saved: {path}  (step={step}, val_loss={val_loss:.4f})")
    return path


def _build_datasets(cfg: TrainConfig):
    """Return (train_ds, val_ds, test_ds) as KlineWindowDataset instances.

    All three share the same kline/funding/liq parquets. The split is by bar
    index of the full kline parquet (72/12/16). Each split uses a separate
    KlineWindowDataset constructed with start/end row slices via a Subset
    wrapper so the underlying array is never copied.
    """
    full_ds = KlineWindowDataset(
        path=cfg.kline_path,
        window=cfg.window,
        stride=1,  # stride=1 so we can slice by window index later
        funding_path=cfg.funding_path,
        liq_path=cfg.liq_path,
        apply_features=True,
        return_targets=True,
    )
    n = len(full_ds)
    # The underlying bar array has (n + window - 1) bars. We re-derive split
    # boundaries from bar count so the 72/12/16 split is done on bars, not windows.
    n_bars = full_ds._bars.shape[0]
    train_end_bar = int(n_bars * cfg.train_frac)
    val_end_bar = int(n_bars * (cfg.train_frac + cfg.val_frac))

    # Window index i corresponds to bars [i*stride, i*stride + window - 1].
    # With stride=1: train windows have start bar in [0, train_end_bar - window].
    def bar_to_window_idx(bar: int) -> int:
        return max(0, bar - cfg.window + 1)

    train_end_win = bar_to_window_idx(train_end_bar)
    val_end_win = bar_to_window_idx(val_end_bar)

    train_indices = list(range(0, train_end_win, cfg.stride_train))
    val_indices = list(range(train_end_win, val_end_win, cfg.stride_val))
    test_indices = list(range(val_end_win, n, cfg.stride_val))

    return (
        Subset(full_ds, train_indices),
        Subset(full_ds, val_indices),
        Subset(full_ds, test_indices),
    )


def _fit_tokenizer(cfg: TrainConfig, train_subset) -> ReturnTokenizerV2:
    """Collect log returns from training windows and fit the tokenizer."""
    log.info("Fitting tokenizer on training split log returns...")
    all_rets = []
    for feats, rets in train_subset:
        all_rets.append(rets.numpy())
    all_rets_arr = np.concatenate(all_rets).astype(np.float64)
    all_rets_arr = all_rets_arr[np.isfinite(all_rets_arr)]
    tok = ReturnTokenizerV2(n_bins=cfg.n_bins)
    tok.fit(all_rets_arr)
    tok.save(cfg.tokenizer_path)
    log.info(f"Tokenizer fitted: n_bins={tok.n_bins}, saved to {cfg.tokenizer_path}")
    return tok


def _collate(batch, tokenizer: ReturnTokenizerV2, device: torch.device):
    """Stack features and convert log returns to bin IDs."""
    feats_list = []
    ids_list = []
    for feats, log_rets in batch:
        feats_list.append(feats)
        ids = tokenizer.encode(log_rets.numpy())
        ids_list.append(torch.from_numpy(ids))
    feats = torch.stack(feats_list).to(device)
    ids = torch.stack(ids_list).to(device)
    return feats, ids


@torch.no_grad()
def _eval_loss(
    model: CandleGPTv2,
    val_subset,
    tokenizer: ReturnTokenizerV2,
    device: torch.device,
    n_batches: int,
    batch_size: int,
) -> float:
    model.eval()
    loader = DataLoader(val_subset, batch_size=batch_size, shuffle=False, drop_last=True)
    losses = []
    for i, batch in enumerate(loader):
        if i >= n_batches:
            break
        feats, ids = _collate(batch, tokenizer, device)
        logits = model(feats)  # (B, T, n_bins)
        B, T, V = logits.shape
        loss = nn.functional.cross_entropy(
            logits.view(B * T, V), ids.view(B * T)
        )
        losses.append(loss.item())
    model.train()
    return float(np.mean(losses)) if losses else float("inf")


def train(cfg: TrainConfig) -> str:
    """Run the training loop. Returns run_id."""
    cfg.run_dir.mkdir(parents=True, exist_ok=True)
    cfg.ckpt_dir.mkdir(parents=True, exist_ok=True)

    logging.basicConfig(
        level=logging.INFO,
        format="%(asctime)s %(levelname)s %(message)s",
        handlers=[
            logging.FileHandler(cfg.run_dir / "train.log"),
            logging.StreamHandler(),
        ],
    )
    log.info(f"=== CandleGPTv2 Training Run: {cfg.run_id} ===")
    log.info(f"run_dir: {cfg.run_dir}")

    device = _select_device()
    log.info(f"Device: {device}")

    train_ds, val_ds, test_ds = _build_datasets(cfg)
    log.info(f"Split sizes — train: {len(train_ds)}, val: {len(val_ds)}, test: {len(test_ds)}")

    tokenizer = _fit_tokenizer(cfg, train_ds)

    model = CandleGPTv2(cfg.model).to(device)
    log.info(f"Model params: {model.num_params():,}")

    optimizer = torch.optim.AdamW(
        model.parameters(),
        lr=cfg.lr_max,
        betas=(cfg.beta1, cfg.beta2),
        weight_decay=cfg.weight_decay,
    )

    train_loader = DataLoader(
        train_ds,
        batch_size=cfg.batch_size,
        shuffle=True,
        drop_last=True,
        num_workers=0,
    )

    best_val_loss = float("inf")
    best_ckpt_path: Path | None = None
    step = 0
    train_start = time.time()
    last_ckpt_time = train_start
    tokens_processed = 0

    log.info("Starting training loop...")
    model.train()

    while True:
        for batch in train_loader:
            # --- Wall-clock cap ---
            elapsed = time.time() - train_start
            if elapsed >= cfg.max_wall_clock_s:
                log.info(
                    f"[STOP] 6-hour wall-clock cap reached at step {step} "
                    f"({elapsed/3600:.2f}h elapsed). Saving final checkpoint."
                )
                tag = f"step_{step:07d}"
                _save_checkpoint(model, optimizer, step, best_val_loss, cfg, tag)
                return cfg.run_id

            if step >= cfg.max_steps:
                log.info(f"[STOP] max_steps={cfg.max_steps} reached.")
                return cfg.run_id

            # --- LR update ---
            lr = _get_lr(step, cfg)
            for pg in optimizer.param_groups:
                pg["lr"] = lr

            # --- Forward + backward ---
            feats, ids = _collate(batch, tokenizer, device)
            logits = model(feats)
            B, T, V = logits.shape
            loss = nn.functional.cross_entropy(
                logits.view(B * T, V), ids.view(B * T)
            )
            optimizer.zero_grad()
            loss.backward()
            nn.utils.clip_grad_norm_(model.parameters(), cfg.grad_clip)
            optimizer.step()

            tokens_processed += B * T
            step += 1

            # --- Logging ---
            if step % cfg.log_interval_steps == 0:
                elapsed_h = (time.time() - train_start) / 3600
                log.info(
                    f"step={step:6d}  loss={loss.item():.4f}  "
                    f"lr={lr:.2e}  elapsed={elapsed_h:.2f}h"
                )

            # --- Validation ---
            if step % cfg.val_interval_steps == 0:
                val_loss = _eval_loss(
                    model, val_ds, tokenizer, device,
                    cfg.val_batches, cfg.batch_size,
                )
                log.info(f"  [val] step={step}  val_loss={val_loss:.4f}")
                if val_loss < best_val_loss:
                    best_val_loss = val_loss
                    best_ckpt_path = _save_checkpoint(
                        model, optimizer, step, val_loss, cfg, "best_val"
                    )
                    log.info(f"  [val] NEW BEST: {val_loss:.4f}")

            # --- Periodic checkpoint ---
            now = time.time()
            if now - last_ckpt_time >= cfg.checkpoint_interval_s:
                tag = f"step_{step:07d}"
                _save_checkpoint(model, optimizer, step, best_val_loss, cfg, tag)
                last_ckpt_time = now

    return cfg.run_id
```

```python
"""Post-training evaluation of the best checkpoint on the held-out test split.

evaluate(cfg, ckpt_path) computes:
  - Overall top-1 accuracy (predicted bin == true bin)
  - Per-regime accuracy (regime 0/1/2 and -1 untagged)
  - Calibration: 10-bucket ECE (expected calibration error)
  - Best val loss (from checkpoint metadata)
  - Sample predictions: 5 random windows, show predicted vs actual returns

write_report(metrics, run_dir) writes REPORT.md in the run directory.
"""
from __future__ import annotations
import json
import logging
import time
from pathlib import Path

import numpy as np
import torch
import torch.nn.functional as F
from torch.utils.data import DataLoader, Subset

from v2.data.dataset import KlineWindowDataset
from v2.features.constants import FEATURE_COLUMNS
from v2.model.model import CandleGPTv2
from v2.model.config import ModelConfig
from v2.model.tokenizer import ReturnTokenizerV2
from v2.train.config import TrainConfig

log = logging.getLogger(__name__)


def _select_device() -> torch.device:
    if torch.backends.mps.is_available():
        return torch.device("mps")
    if torch.cuda.is_available():
        return torch.device("cuda")
    return torch.device("cpu")


def _collate(batch, tokenizer: ReturnTokenizerV2, device: torch.device):
    feats_list, ids_list = [], []
    for feats, log_rets in batch:
        feats_list.append(feats)
        ids_list.append(torch.from_numpy(tokenizer.encode(log_rets.numpy())))
    return torch.stack(feats_list).to(device), torch.stack(ids_list).to(device)


@torch.no_grad()
def evaluate(cfg: TrainConfig, ckpt_path: Path) -> dict:
    """Evaluate best checkpoint on the test split. Returns metrics dict."""
    device = _select_device()

    # Load checkpoint
    ckpt = torch.load(ckpt_path, map_location="cpu", weights_only=False)
    model_cfg = ModelConfig.from_dict(ckpt["model_config"])
    model = CandleGPTv2(model_cfg).to(device)
    model.load_state_dict(ckpt["model_state"])
    model.eval()
    log.info(f"Loaded checkpoint: {ckpt_path}  step={ckpt['step']}")

    # Load tokenizer
    tok = ReturnTokenizerV2.load(cfg.tokenizer_path)

    # Build test dataset (same full_ds, test indices)
    full_ds = KlineWindowDataset(
        path=cfg.kline_path,
        window=cfg.window,
        stride=1,
        funding_path=cfg.funding_path,
        liq_path=cfg.liq_path,
        apply_features=True,
        return_targets=True,
    )
    n_bars = full_ds._bars.shape[0]
    train_end_bar = int(n_bars * cfg.train_frac)
    val_end_bar = int(n_bars * (cfg.train_frac + cfg.val_frac))

    def bar_to_window_idx(bar: int) -> int:
        return max(0, bar - cfg.window + 1)

    val_end_win = bar_to_window_idx(val_end_bar)
    test_indices = list(range(val_end_win, len(full_ds), cfg.stride_val))
    test_ds = Subset(full_ds, test_indices)
    log.info(f"Test windows: {len(test_ds)}")

    # Regime column index in the feature vector
    regime_0_idx = FEATURE_COLUMNS.index("regime_0")
    regime_1_idx = FEATURE_COLUMNS.index("regime_1")
    regime_2_idx = FEATURE_COLUMNS.index("regime_2")

    loader = DataLoader(test_ds, batch_size=cfg.batch_size, shuffle=False, drop_last=False)

    all_correct = []
    all_probs_top1 = []  # confidence of predicted bin
    regime_stats = {r: {"correct": 0, "total": 0} for r in [-1, 0, 1, 2]}
    sample_windows = []

    for batch_idx, batch in enumerate(loader):
        feats, ids = _collate(batch, tok, device)  # (B, T, 41), (B, T)
        logits = model(feats)  # (B, T, n_bins)
        probs = F.softmax(logits, dim=-1)
        pred_ids = logits.argmax(dim=-1)  # (B, T)
        correct = (pred_ids == ids)  # (B, T) bool

        # Flatten
        correct_flat = correct.view(-1).cpu().numpy()
        probs_flat = probs.view(-1, tok.n_bins).cpu().numpy()
        ids_flat = ids.view(-1).cpu().numpy()
        feats_flat = feats.view(-1, feats.size(-1)).cpu().numpy()

        all_correct.append(correct_flat)

        # Confidence of the predicted bin
        pred_ids_flat = pred_ids.view(-1).cpu().numpy()
        top1_conf = probs_flat[np.arange(len(pred_ids_flat)), pred_ids_flat]
        all_probs_top1.append(top1_conf)

        # Per-regime accuracy
        r0 = feats_flat[:, regime_0_idx] > 0.5
        r1 = feats_flat[:, regime_1_idx] > 0.5
        r2 = feats_flat[:, regime_2_idx] > 0.5
        r_none = ~(r0 | r1 | r2)
        for mask, r_id in [(r0, 0), (r1, 1), (r2, 2), (r_none, -1)]:
            if mask.any():
                regime_stats[r_id]["correct"] += int(correct_flat[mask].sum())
                regime_stats[r_id]["total"] += int(mask.sum())

        # Collect first 5 windows for sample predictions
        if len(sample_windows) < 5:
            b = 0  # first item in batch
            sample_windows.append({
                "true_ids": ids[b, :10].cpu().tolist(),
                "pred_ids": pred_ids[b, :10].cpu().tolist(),
                "true_rets": tok.decode(ids[b, :10].cpu().numpy()).tolist(),
                "pred_rets": tok.decode(pred_ids[b, :10].cpu().numpy()).tolist(),
            })

    all_correct_arr = np.concatenate(all_correct)
    all_probs_arr = np.concatenate(all_probs_top1)

    overall_accuracy = float(all_correct_arr.mean())

    # Calibration: ECE over 10 confidence buckets
    n_buckets = 10
    bucket_edges = np.linspace(0.0, 1.0, n_buckets + 1)
    ece_buckets = []
    for lo, hi in zip(bucket_edges[:-1], bucket_edges[1:]):
        mask = (all_probs_arr >= lo) & (all_probs_arr < hi)
        if mask.sum() > 0:
            avg_conf = float(all_probs_arr[mask].mean())
            avg_acc = float(all_correct_arr[mask].mean())
            frac = float(mask.sum()) / len(mask)
            ece_buckets.append({"lo": lo, "hi": hi, "conf": avg_conf,
                                 "acc": avg_acc, "frac": frac})
    ece = float(sum(b["frac"] * abs(b["conf"] - b["acc"]) for b in ece_buckets))

    per_regime = {}
    for r_id, stats in regime_stats.items():
        if stats["total"] > 0:
            per_regime[str(r_id)] = {
                "accuracy": stats["correct"] / stats["total"],
                "n": stats["total"],
            }

    return {
        "run_id": cfg.run_id,
        "ckpt_path": str(ckpt_path),
        "ckpt_step": ckpt.get("step"),
        "best_val_loss": ckpt.get("val_loss"),
        "overall_accuracy": overall_accuracy,
        "ece": ece,
        "calibration_buckets": ece_buckets,
        "per_regime_accuracy": per_regime,
        "sample_predictions": sample_windows,
        "n_test_bars": int(all_correct_arr.shape[0]),
    }


def write_report(metrics: dict, run_dir: Path) -> None:
    """Write REPORT.md in the run directory."""
    run_dir = Path(run_dir)
    run_dir.mkdir(parents=True, exist_ok=True)
    lines = [
        f"# CandleGPTv2 Training Report — {metrics['run_id']}",
        "",
        f"**Checkpoint:** step {metrics['ckpt_step']}  ",
        f"**Best val loss:** {metrics['best_val_loss']:.4f}  ",
        f"**Test bars evaluated:** {metrics['n_test_bars']:,}  ",
        "",
        "## Accuracy",
        "",
        f"- Overall top-1 bin accuracy: **{metrics['overall_accuracy']:.4f}** "
        f"({metrics['overall_accuracy']*100:.2f}%)",
        f"- ECE (10-bucket): **{metrics['ece']:.4f}**",
        "",
        "## Per-Regime Accuracy",
        "",
        "| Regime | Accuracy | N bars |",
        "|--------|----------|--------|",
    ]
    regime_names = {"-1": "Untagged", "0": "Regime 0", "1": "Regime 1", "2": "Regime 2"}
    for r_id, stats in sorted(metrics["per_regime_accuracy"].items()):
        name = regime_names.get(r_id, f"Regime {r_id}")
        lines.append(
            f"| {name} | {stats['accuracy']:.4f} ({stats['accuracy']*100:.1f}%) "
            f"| {stats['n']:,} |"
        )

    lines += [
        "",
        "## Calibration (10 confidence buckets)",
        "",
        "| Conf range | Avg conf | Avg acc | Fraction |",
        "|-----------|---------|---------|----------|",
    ]
    for b in metrics["calibration_buckets"]:
        lines.append(
            f"| [{b['lo']:.1f}, {b['hi']:.1f}) "
            f"| {b['conf']:.3f} | {b['acc']:.3f} | {b['frac']:.3f} |"
        )

    lines += [
        "",
        "## Sample Predictions (first 10 positions of first 5 test windows)",
        "",
        "Each row: predicted return | actual return (log-return, e.g. 0.0012 = +0.12%)",
        "",
    ]
    for i, s in enumerate(metrics["sample_predictions"]):
        lines.append(f"**Window {i+1}:**")
        rows = ["| # | Predicted | Actual |", "|---|---------|--------|"]
        for j, (p, a) in enumerate(zip(s["pred_rets"], s["true_rets"])):
            rows.append(f"| {j} | {p:+.5f} | {a:+.5f} |")
        lines.extend(rows)
        lines.append("")

    report = "\n".join(lines)
    report_path = run_dir / "REPORT.md"
    report_path.write_text(report, encoding="utf-8")
    log.info(f"Report written: {report_path}")
```

## Task 4: CLI entry point — `run.py`

- Create: `v2/train/run.py`

- [ ] **Step 1: Implement `run.py`**

Path: `v2/train/run.py`:

```python
"""CLI entry point for CandleGPTv2 training.

Usage:
    cd projects/candle-gpt
    nohup uv run python -m v2.train.run [--run-id ID] [--batch-size N] \
        > v2/runs/latest_launch.log 2>&1 &
    echo $! > v2/runs/train.pid

The process writes its own detailed log to v2/runs/<run_id>/train.log.
After training finishes (or hits the 6-hour cap), it automatically evaluates
the best checkpoint and writes REPORT.md in the run directory.
"""
from __future__ import annotations
import argparse
import logging
import sys
from pathlib import Path

from v2.train.config import TrainConfig
from v2.train.loop import train
from v2.train.eval import evaluate, write_report

log = logging.getLogger(__name__)


def main() -> None:
    ap = argparse.ArgumentParser(description="Train CandleGPTv2")
    ap.add_argument("--run-id", type=str, default=None,
                    help="Run ID (default: timestamp)")
    ap.add_argument("--batch-size", type=int, default=32)
    ap.add_argument("--max-steps", type=int, default=200_000)
    ap.add_argument("--lr-max", type=float, default=3e-4)
    ap.add_argument("--window", type=int, default=512)
    ap.add_argument("--raw-dir", type=Path,
                    default=Path("v2/data/raw"))
    ap.add_argument("--runs-dir", type=Path,
                    default=Path("v2/runs"))
    ap.add_argument("--eval-only", type=str, default=None,
                    help="Skip training; evaluate this checkpoint path directly")
    args = ap.parse_args()

    cfg = TrainConfig(
        raw_dir=args.raw_dir,
        runs_dir=args.runs_dir,
        batch_size=args.batch_size,
        max_steps=args.max_steps,
        lr_max=args.lr_max,
        window=args.window,
    )
    if args.run_id:
        cfg.run_id = args.run_id

    if args.eval_only:
        cfg.run_dir.mkdir(parents=True, exist_ok=True)
        metrics = evaluate(cfg, Path(args.eval_only))
        write_report(metrics, cfg.run_dir)
        print(f"Evaluation complete. Report: {cfg.report_path}")
        return

    print(f"Starting training run: {cfg.run_id}")
    print(f"Run dir: {cfg.run_dir}")
    print(f"Max wall clock: {cfg.max_wall_clock_s/3600:.1f}h")

    run_id = train(cfg)
    print(f"Training finished. run_id={run_id}")

    # Evaluate best checkpoint
    if cfg.best_ckpt_path.exists():
        print(f"Evaluating best checkpoint: {cfg.best_ckpt_path}")
        metrics = evaluate(cfg, cfg.best_ckpt_path)
        write_report(metrics, cfg.run_dir)
        print(f"Report: {cfg.report_path}")
        print(f"Overall accuracy: {metrics['overall_accuracy']:.4f}")
        print(f"ECE: {metrics['ece']:.4f}")
        for r_id, stats in sorted(metrics["per_regime_accuracy"].items()):
            print(f"  Regime {r_id}: acc={stats['accuracy']:.4f}  n={stats['n']:,}")
    else:
        print("WARNING: No best_val checkpoint found — evaluation skipped.")


if __name__ == "__main__":
    main()
```
